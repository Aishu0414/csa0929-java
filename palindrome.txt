import java.util.Scanner; 
class Palindrome { 
    public static void main(String args[]){
    int n = 121;
    int temp = n, rev = 0;
    while (n != 0) {
        int d = n % 10;
         rev = rev * 10 + d; 
         n /= 10; 
    } 
    if (rev == temp){ 
        System.out.println(temp + " is a Palindrome.");
    }else{ System.out.println(temp + " is NOT a Palindrome.");
    } 
    } 
}
